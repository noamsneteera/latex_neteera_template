\documentclass{beamer}

% Theme and color
\usetheme{Warsaw}
\usepackage{xcolor}
\graphicspath{{../latex_neteera_template/}{../../latex_neteera_template/}}
\definecolor{neteeraBG}{RGB}{248,248,248}
\definecolor{bannercolor}{RGB}{86,114,127}

\setbeamercolor{structure}{fg=bannercolor, bg=neteeraBG}
\setbeamercolor{background canvas}{fg=bannercolor, bg=neteeraBG}

% Other packages
\usepackage{graphicx}
\usepackage{tikz}
\usepackage[absolute,overlay]{textpos}

% Metadata
\title[\shorttitle]{\longtitle}
\author{\myname}
\date{\whichdate}

% Custom title slide with background image and bannercolor text
\makeatletter
\defbeamertemplate*{title page}{bg image positioned}
{
  \begin{tikzpicture}[remember picture,overlay]
    \node[at=(current page.center), inner sep=0] {
      \includegraphics[width=\paperwidth,height=\paperheight]{neteera_title.png}
    };
  \end{tikzpicture}

  \begin{textblock*}{10cm}(2cm,6cm)
    {\usebeamerfont{title}\color{bannercolor}\inserttitle\par}
    \vskip0.3cm
    {\usebeamerfont{author}\color{bannercolor}\insertauthor\par}
    \vskip0.2cm
    {\usebeamerfont{date}\color{bannercolor}\insertdate\par}
  \end{textblock*}
}
\makeatother

% Logo and banner only on normal slides
\addtobeamertemplate{background}{
    \ifbool{beamer@plainframe}{}{
        \begin{textblock*}{30pt}(0.02\paperwidth,0.855\paperheight)
            \includegraphics[height=30pt]{neteera_logo.png}
        \end{textblock*}
        \begin{textblock*}{0.8\paperwidth}(0.17\paperwidth,0.92\paperheight)
            \color{bannercolor}\rule{\textwidth}{0.3pt}
        \end{textblock*}
        \begin{textblock*}{0.8\paperwidth}(0.17\paperwidth,0.93\paperheight)
            \color{bannercolor}\tiny Confidential \& Proprietary
        \end{textblock*}
    }
}{}
